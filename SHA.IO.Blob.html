Module SHA.IO.Blob

module Blob: SHA.H  with type t = Blob.t
include SHA.S
include Object.IO
val of_hex : string -> t
Convert an hex-encoded string into a sha1 value. Raise SHA.Ambiguous if the hash is short; in that case use SHA.H.of_short_hex.
val of_short_hex : string -> t
Same as SHA.H.of_hex but allow values shorter than 20 characters. Such hash values are called short hashes and can cause some functions to raise SHA.Ambiguous.
val input_hex : Mstruct.t -> t
Read an hex-encode SHA value.
val add_hex : Buffer.t -> t -> unit
Add the hex-encoding of the SHA value to the buffer.
val is_short : t -> bool
Check if the SHA is short.
val zero : t
A SHA full of zero. Useful for padding.
