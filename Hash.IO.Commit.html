Module Hash.IO.Commit

module Commit: Hash.H  with type t = Commit.t
include Hash.S
include Object.IO
val of_hex : string -> t
Convert an hex-encoded string into a hash value. Raise Hash.Ambiguous if the hash is short; in that case use Hash.H.of_short_hex.
val of_short_hex : string -> t
Same as Hash.H.of_hex but allow values shorter than 20 characters. Such hash values are called short hashes and can cause some functions to raise Hash.Ambiguous.
val input_hex : Mstruct.t -> t
Read an hex-encoded hash value.
val add_hex : Buffer.t -> t -> unit
Add the hex-encoding of the hash value to the buffer.
val is_short : t -> bool
Check if the hash is short.
val zero : t
A hash full of zero. Useful for padding.
