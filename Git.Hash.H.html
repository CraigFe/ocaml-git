Module type Git.Hash.H

module type H = sig .. end
include Git.Hash.S
include Git.IO
val of_hex : string -> t
Convert an hex-encoded string into a hash value. Raise Git.Hash.Ambiguous if the hash is short; in that case use Git.Hash.H.of_short_hex.
val of_short_hex : string -> t
Same as Git.Hash.H.of_hex but allow values shorter than 20 characters. Such hash values are called short hashes and can cause some functions to raise Git.Hash.Ambiguous.
val input_hex : Mstruct.t -> t
Read an hex-encoded hash value.
val add_hex : Buffer.t -> t -> unit
Add the hex-encoding of the hash value to the buffer.
val is_short : t -> bool
Check if the hash is short.
val zero : t
A hash full of zero. Useful for padding.
